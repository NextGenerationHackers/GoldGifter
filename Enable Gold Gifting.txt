    var variable_cost_params = {};

    function hideTopupWithLogging(additional_data)
    {
        props = {
            'tag' : 'bonus_gold',
            'action' : 'close_window',
            'gold_purchased' : 0
        };

        if(additional_data != null && typeof additional_data === 'object')
        {
            var props = jQuery.extend(props, additional_data);
        }

        cc.logGenericEvent(props);
        cc.hideTopup();
    }
